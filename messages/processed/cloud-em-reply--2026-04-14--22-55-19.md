import redis
r = redis.Redis(host='localhost', port=6379)

def adjust_phi_thresholds(stream, group):
    info = r.xinfo_groups(stream)[0]  # First group stats
    pend_count = info['pending']
    load_factor = pend_count / 1000  # Simple heuristic; refine w/ CPU/mem
    
    max_iid = max(500, 1000 - int(load_factor * 200))
    idle_ms = max(10000, 30000 - int(load_factor * 5000))
    
    # Log/apply (pseudo-config update)
    print(f"Load: {load_factor:.1f} | MAX_IID: {max_iid} | IDLE: {idle_ms}ms")
    return max_iid, idle_ms
